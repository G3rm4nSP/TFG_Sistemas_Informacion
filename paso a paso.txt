lo siguiente es generar un script que sirva como instalador que descrge del hup de Docker el front y el back para instalarlo en la raspberry asi como prepararla para kiok que es que se inicie siempre en la aplicacion en pantalla completa


creando la imagen de raspi SO lite para que sea plug an play, desde un ordenador Linux añadimos un script que se carga al inicio:

# 1. Descarga la imagen Lite y descomprime
wget https://downloads.raspberrypi.org/raspios_lite_armhf_latest -O raspios_lite.img.xz
xz -d raspios_lite.img.xz

# 2. Monta la imagen
sudo losetup -Pf --show raspios_lite.img
# Verás algo como /dev/loop0

# 3. Monta root y boot
sudo mkdir -p /mnt/rpi-root /mnt/rpi-boot
sudo mount /dev/loop0p2 /mnt/rpi-root
sudo mount /dev/loop0p1 /mnt/rpi-boot

# 4. Copia el script de primer arranque
sudo mkdir -p /mnt/rpi-root/usr/local/bin
sudo cp first_setup.sh /mnt/rpi-root/usr/local/bin/
sudo chmod +x /mnt/rpi-root/usr/local/bin/first_setup.sh

# 5. Configura rc.local para ejecutarlo al inicio
sudo bash -c 'echo "/usr/local/bin/first_setup.sh" >> /mnt/rpi-root/etc/rc.local'

# 6. Desmonta todo y libera loop
sudo umount /mnt/rpi-root /mnt/rpi-boot
sudo losetup -d loop0



"first_setup.sh": 

#!/bin/bash
set -e

# --- Detectar si ya se ejecutó ---
FLAG_FILE="/etc/first_setup_done"
if [ -f "$FLAG_FILE" ]; then
    exit 0
fi

echo "🚀 Configuración inicial de Mini ERP Plug & Play"

# --- 1. Configurar Wi-Fi de forma interactiva ---
read -p "Introduce el SSID de tu red Wi-Fi: " WIFI_SSID
read -s -p "Introduce la contraseña de tu red Wi-Fi: " WIFI_PASS
echo
cat <<EOF > /etc/wpa_supplicant/wpa_supplicant.conf
ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
update_config=1
country=ES

network={
    ssid="$WIFI_SSID"
    psk="$WIFI_PASS"
}
EOF

wpa_cli -i wlan0 reconfigure
echo "✅ Wi-Fi configurado."

# --- 2. Configurar zona horaria e idioma ---
dpkg-reconfigure tzdata
dpkg-reconfigure locales

# --- 3. Actualizar e instalar dependencias mínimas ---
apt-get update
apt-get upgrade -y

apt-get install -y docker.io docker-compose xserver-xorg xinit openbox chromium-browser

# --- 4. Habilitar Docker ---
systemctl enable docker
systemctl start docker
usermod -aG docker $USER

# --- 5. Crear carpeta de la app y docker-compose ---
APP_DIR="/home/$USER/erp-rpi"
mkdir -p $APP_DIR
cd $APP_DIR

cat > docker-compose.yml <<'EOF'
version: "3.9"
services:
  backend:
    image: g3rm4nsp/erp-backend:latest
    container_name: backend
    ports:
      - "3000:3000"
    restart: always

  frontend:
    image: g3rm4nsp/erp-frontend:latest
    container_name: frontend
    ports:
      - "5173:5173"
    environment:
      - VITE_API_URL=http://localhost:3000
    restart: always
EOF

docker compose pull
docker compose up -d

# --- 6. Configurar modo kiosk ---
mkdir -p /home/$USER/.config/lxsession/LXDE-pi
cat > /home/$USER/.config/lxsession/LXDE-pi/autostart <<EOF
@xset s off
@xset -dpms
@xset s noblank
@chromium-browser --noerrdialogs --kiosk http://localhost:5173 --incognito
EOF

# --- 7. Marcar como completado para que no se vuelva a ejecutar ---
touch $FLAG_FILE

echo "✅ Configuración completa. Reinicia tu Raspberry Pi para empezar a usar Mini ERP Plug & Play."
